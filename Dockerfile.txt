# اختيار نسخة Node.js المناسبة لمشروعك
FROM node:18-alpine

# تحديد مجلد العمل داخل السيرفر
WORKDIR /usr/src/app

# نسخ ملفات الحزم وتثبيتها
COPY package*.json ./
RUN npm install --only=production

# نسخ بقية ملفات اللعبة والسيرفر
COPY . .

# فتح المنفذ الذي سيعمل عليه السيرفر (Google Cloud Run يحدد المنفذ تلقائياً عبر متغير PORT)
EXPOSE 8080

# أمر تشغيل اللعبة
CMD [ "node", "server.js" ]
